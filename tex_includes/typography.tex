% tex_includes/typography.tex
% ------------------------------------------------------------
% Typography + heading styling for resume builds (ATS + Color)
%
% Notes:
% - We use Latin Modern fonts (ATS-safe, widely available, predictable).
% - Heading formatting/spacing is controlled by titlesec.
% - We suppress Pandoc's default title/author block.
% ------------------------------------------------------------

\usepackage{titlesec}
\usepackage{anyfontsize}

% Font selection (ATS-safe / predictable)
% ------------------------------------------------------------
% Despite older comments elsewhere, this file does NOT set Arial.
% We intentionally use Latin Modern for consistency and portability.
\usepackage{fontspec}
\setmainfont{Latin Modern Roman}
\setsansfont{Latin Modern Sans}
\setmonofont{Latin Modern Mono}

% Base body font size + line height
% ------------------------------------------------------------
% Body text: 10pt font with 12pt baseline, then a slight global line spread.
\AtBeginDocument{\fontsize{10pt}{12pt}\selectfont}
\linespread{1.05}\selectfont

% Heading styles (titlesec)
% ------------------------------------------------------------
% Section (H1), Subsection (H2), Paragraph (subheader-ish)
\titleformat{\section}{\bfseries\fontsize{13pt}{15pt}\selectfont}{}{0pt}{}
\titleformat{\subsection}{\bfseries\fontsize{11pt}{13pt}\selectfont}{}{0pt}{}
\titleformat{\paragraph}{\bfseries\fontsize{10pt}{12pt}\selectfont}{}{0pt}{}

% Heading spacing (left indent, space before, space after)
% ------------------------------------------------------------
% Keep this tight but readable for a one-page resume layout.
\titlespacing*{\section}{0pt}{6pt}{4pt}
\titlespacing*{\subsection}{0pt}{4pt}{2pt}
\titlespacing*{\paragraph}{0pt}{2pt}{1pt}

% Suppress Pandoc title/author blocks
% ------------------------------------------------------------
% Pandoc's default LaTeX template may emit \author{} and call \maketitle.
% We blank them so nothing prints.
\makeatletter
\def\@author{}
\def\maketitle{}
\makeatother
